202  template<class A>
203  inline Variable& Assign(atl::Tape<REAL_T>& tape, const ExpressionBase<REAL_T, A>& exp, size_t index) {
204 
205  atl::StackEntry<REAL_T>& entry = tape.stack[index];
206  exp.PushIds(entry.ids);
207  entry.w = this->info;
208  entry.first.resize(entry.ids.size());
209  typename atl::StackEntry<REAL_T>::vi_iterator it;
210  typename atl::StackEntry<REAL_T>::vi_iterator jt;
211  size_t i = 0;
212  size_t j = 0;
213  switch (tape.derivative_trace_level) {
214 
215  case FIRST_ORDER_REVERSE:
216  for (it = entry.ids.begin(); it != entry.ids.end(); ++it) {
217  entry.first[i] = exp.EvaluateDerivative((*it)->id);
218  i++;
219  }
220  break;
221 
222  case SECOND_ORDER_REVERSE:
223  entry.second.resize(entry.ids.size() * entry.ids.size());
224  for (it = entry.ids.begin(); it != entry.ids.end(); ++it) {
225 
226  entry.first[i] = exp.EvaluateDerivative((*it)->id);
227  j = 0;
228  for (jt = entry.ids.begin(); jt != entry.ids.end(); ++jt) {
229 
230  entry.second[i*entry.ids.size()+j] = exp.EvaluateDerivative((*it)->id, (*jt)->id);
231 
232  j++;
233  }
234  i++;
235  }
236 
237  break;
238 
239  case THIRD_ORDER_REVERSE:
240  break;
241 
242  default:
243  std::cout << "Unknown Derivative Trace Level.\n";
244  exit(0);
245  }
246 
247  this->info->value = exp.GetValue();
248  return *this;
249  }
