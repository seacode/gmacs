295  const T nonparametric(const T &x, const T &selparms)
296  {
297  int x2 = x.indexmax();
298  dvar_vector selex(1,x2);
299  for (int i=1; i<=x2; i++)
300  selex(i) = (1.0)/(1.0+mfexp(selparms(i)));
301  dvariable temp = selex(x2);
302  selex /= temp;
303  return selex;
